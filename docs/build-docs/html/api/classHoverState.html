class HoverState

HoverState - Mouse hover state management.

Tracks hover state for a component and provides callbacks for state changes. Handles mouse cursor changes and hover-based styling.

Usage: class MyComponent : public juce::Component { HoverState hoverState;

MyComponent() { hoverState.onHoverChanged = [this](bool h) { repaint(); }; }

void mouseEnter(const juce::MouseEvent&) override { hoverState.setHovered(true); } void mouseExit(const juce::MouseEvent&) override { hoverState.setHovered(false); }

void paint(juce::Graphics& g) override { g.setColour(hoverState.isHovered() ? hoverColor : normalColor); } };
